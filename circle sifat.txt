#include <GL/glut.h>

int x_center = 0;
int y_center = 0;
int r = 150;

void drawPixel(int x, int y)
{
    glBegin(GL_POINTS);
    glVertex2i(x_center + x, y_center + y);
    glVertex2i(x_center - x, y_center + y);
    glVertex2i(x_center + x, y_center - y);
    glVertex2i(x_center - x, y_center - y);
    glVertex2i(x_center + y, y_center + x);
    glVertex2i(x_center - y, y_center + x);
    glVertex2i(x_center + y, y_center - x);
    glVertex2i(x_center - y, y_center - x);
    glEnd();
}

void midpointCircle()
{
    int x = 0;
    int y = r;
    int d = 1 - r;
    drawPixel(x, y);
    while (x < y)
    {
        x++;
        if (d < 0)
            d = d + 2 * x + 1;
        else
        {
            y--;
            d = d + 2 * (x - y) + 1;
        }
        drawPixel(x, y);
    }
    glFlush();
}
void display()
{
    glClear(GL_COLOR_BUFFER_BIT);
    midpointCircle();
}

void init()
{
    glClearColor(0.0, 0.0, 0.0, 0.0);
    glColor3f(1.0, 1.0, 0.0);
    glPointSize(2.0);
    gluOrtho2D(-500, 500, -500, 500);
}

int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(600, 600);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Midpoint Circle Drawing Algorithm - Yellow Circle");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
